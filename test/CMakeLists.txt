include(DownloadProject.cmake)
download_project(PROJ                googletest

                 GIT_TAG             master

)

set(TST_SOURCES
	tst_example.cpp
)

add_executable(example_test ${TST_SOURCES})

target_link_libraries(example_test gtest gmock_main)
